    // Format Rp pakai titik
    function formatRupiah(value) {
      const numberString = String(value).replace(/[^0-9]/g, '');
      if (!numberString) return '';
      return 'Rp ' + numberString.replace(/\B(?=(\d{3})+(?!\d))/g, '.');
    }

    // (Opsional) bisa dipakai nanti kalau mau auto-fill dari paket
    function fillForm(item, price) {
      const orderItemInput = document.getElementById('orderItem');
      const orderPriceInput = document.getElementById('orderPrice');
      if (orderItemInput) orderItemInput.value = item;
      if (orderPriceInput) orderPriceInput.value = formatRupiah(price);
    }

    // Kirim form ke WhatsApp
    function sendToWhatsApp(e) {
      e.preventDefault();
      const userId = document.getElementById('userId').value.trim();
      const nickname = document.getElementById('nickname').value.trim();
      const item = document.getElementById('orderItem').value.trim();
      const price = document.getElementById('orderPrice').value.trim();

      const message =
        'Order MLBB\n' +
        'User ID (Server): ' + userId + '\n' +
        'Nickname in game: ' + nickname + '\n' +
        'Jumlah order: ' + item + '\n' +
        'Harga: ' + price;

      const phone = '6283197962700'; // dari 0831... ke format internasional
      const url = 'https://wa.me/' + phone + '?text=' + encodeURIComponent(message);
      window.open(url, '_blank');
    }

    // Auto-format input harga saat user ketik
    document.addEventListener('DOMContentLoaded', function () {
      const priceInput = document.getElementById('orderPrice');
      if (priceInput) {
        priceInput.addEventListener('input', function () {
          this.value = formatRupiah(this.value);
        });
      }
    });
